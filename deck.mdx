function Counter() {
  const [count, setCount] = React.useState(0);

  function increment() {
    setCount(prevCount => prevCount + 1);
  }

  function decrement() {
    setCount(prevCount => prevCount - 1);
  }

  return (
    <div style={{ marginTop: 20 }}>
      <button type="button" onClick={increment}>Increment</button>
      {` ${count} `}
      <button type="button" onClick={decrement}>Decrement</button>
    </div>
  );
}
